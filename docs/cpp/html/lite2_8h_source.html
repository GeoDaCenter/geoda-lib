35inline double ComputeMean(double *dt, int dim) {
36 double sdt = 0.0;
37 for (int i = 0; i < dim; i++) sdt += dt[i];
38 sdt = sdt / ((double)dim);
39 return sdt;
40}

42inline double ComputeSdev(double *dt, int dim) {
43 double mean = ComputeMean(dt, dim);
44 double sum = 0;
45 double diff = 0;
46 for (int cnt = 0; cnt < dim; ++cnt) {
47 diff = dt[cnt] - mean;
48 sum += (diff * diff);
49 }
50 return sqrt(sum / ((double)dim));
51}

490/*
491 HeapSort
492 Sorts vector in the ascending order, using HeapSort.
493 Assumes operator < is defined.
494 For basic idea see Numerical Recipes in C., p.247.
495 */
496template <class T>
497void HeapSort(Iterator<T> v) {
498 const size_t half = v.count() >> 1;
499 if (half == 0) return; // nothing to sort;
500 T *start = v.first();
501 T *hire = start + half, *boss, *empl;
502 // hire will be decremented down to begin during the 'hiring' (heap creation).
503 T *ir = v.last();
504 T rra;
505 while (ir > start) {
506 if (hire > start)
507 rra = *(--hire); // still hiring
508 else { // retirement and promotion
509 rra = *ir; // clear space at end of vector
510 *ir-- = *start; // retire the top of the heap into it
511 }
512 boss = hire; // set up to sift down rra to its proper level
513 empl = boss + (boss - start);
514 while (++empl <= ir) {
515 if (empl < ir && *empl < *(empl + 1)) ++empl;
516 if (rra < *empl) { // demote rra
517 *boss = *empl;
518 boss = empl;
519 empl = boss + (boss - start);
520 } else
521 empl = ir; // rra is already in place
522 }
523 *boss = rra; // i is right place for rra
524 }
525 return;
526}
